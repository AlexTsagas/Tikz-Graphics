%% Midline Label
\newcommand{\midlinelabel}[3]{
   \node (midlabel) at ($ (#1)!.5!(#2) $) {#3};
   \draw[stealth-] (#1) --  (midlabel);
   \draw[-stealth] (midlabel) -- (#2);
}
%
\newcommand{\midlinelabell}[3]{
   \node[fill = white, inner sep = 0.2pt, outer sep=3pt] (midlabel) at ($ (#1)!.5!(#2) $) {#3};
   \draw[stealth-] (#1) --  (midlabel);
   \draw[-stealth] (midlabel) -- (#2);
}
